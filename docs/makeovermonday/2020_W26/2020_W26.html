def show_chart(country):
    temp_df = df.groupby(['Country', 'WBL Report Year'], as_index=False).mean()
    temp_df_A = temp_df[(temp_df['Country'] == 'Japan')]
    temp_df_B = temp_df[(temp_df['Country'] == country)]
    plt.plot(temp_df_A['WBL Report Year'], temp_df_A['Index Score'])
    plt.plot(temp_df_B['WBL Report Year'], temp_df_B['Index Score'])
    plt.show()
