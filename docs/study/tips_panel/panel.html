[35]:

@pn.depends(select1.param.value, select2.param.value)
def plot(feature1, feature2):
    fig = Figure()
    FigureCanvas(fig)
    ax = fig.add_subplot()
    ax.scatter(df[feature1], df[feature2])
    return fig
